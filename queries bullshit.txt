SELECT countryName, 
	   beer
FROM countryreference c 
LEFT OUTER JOIN Alcohol a ON c.incountryId = a.excountryid 
WHERE a.beer > a.wine AND a.beer > a.spirits
	  AND region = 'South Asia'
ORDER BY a.beer DESC

--wine
SELECT countryName, 
	   wine
FROM countryreference c
LEFT JOIN alcohol a ON c.incountryId=a.excountryid
WHERE a.wine > a.beer AND a.wine > a.spirits
	--user inputs region
	AND region=''
	
--spirits
SELECT countryName, 
	   spirits
FROM countryreference c
LEFT OUTER JOIN alcohol a ON c.incountryId=a.excountryid
WHERE a.spirits > a.beer AND a.spirits > a.wine
	--user inputs region
	AND region=''
